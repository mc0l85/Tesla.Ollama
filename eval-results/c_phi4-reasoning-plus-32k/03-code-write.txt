def merge_sorted(list1, list2):
    """
    Merge two sorted lists of integers into a single sorted list.
    
    Parameters:
        list1 (list of int): The first sorted list.
        list2 (list of int): The second sorted list.
    
    Returns:
        list of int: A merged sorted list containing all elements from list1 and list2.
    """
    i, j = 0, 0
    merged = []
    
    # Compare elements from both lists and add the smaller one to the merged list.
    while i < len(list1) and j < len(list2):
        if list1[i] <= list2[j]:
            merged.append(list1[i])
            i += 1
        else:
            merged.append(list2[j])
            j += 1
    
    # Append any remaining elements from list1 or list2.
    if i < len(list1):
        merged.extend(list1[i:])
    if j < len(list2):
        merged.extend(list2[j:])
    
    return merged

# Test cases
assert merge_sorted([1, 3, 5], [2, 4, 6]) == [1, 2, 3, 4, 5, 6]
assert merge_sorted([10, 20, 30], [5, 15, 25, 35]) == [5, 10, 15, 20, 25, 30, 35]
