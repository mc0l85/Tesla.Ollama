def merge_sorted(list1, list2):
    """
    Merges two sorted lists into one sorted list without using the built-in `sorted()` function or `.sort()` method.
    
    Parameters:
    list1 (list): A list of integers sorted in ascending order.
    list2 (list): A list of integers sorted in ascending order.
    
    Returns:
    list: A new list containing all elements from both input lists, sorted in ascending order.
    """
    merged = []
    i, j = 0, 0

    # Traverse both lists and merge them
    while i < len(list1) and j < len(list2):
        if list1[i] < list2[j]:
            merged.append(list1[i])
            i += 1
        else:
            merged.append(list2[j])
            j += 1

    # Add any remaining elements from list1 or list2
    merged.extend(list1[i:])
    merged.extend(list2[j:])

    return merged


# Test cases
assert merge_sorted([1, 3, 5], [2, 4, 6]) == [1, 2, 3, 4, 5, 6]
assert merge_sorted([10, 20, 30], [5, 15, 25]) == [5, 10, 15, 20, 25, 30]
