def merge_sorted(a, b):
    """
    Merge two sorted lists of integers into a single sorted list.

    The function does **not** use the built‑in `sorted()` or the `.sort()` method; it
    performs an explicit merge by comparing the current elements of the two
    lists.

    Parameters
    ----------
    a : list
        First sorted list of integers.
    b : list
        Second sorted list of integers.

    Returns
    -------
    list
        A new list containing all elements from `a` and `b`, sorted in ascending
        order.

    Examples
    -------
    >>> merge_sorted([1, 3, 5], [2, 4, 6])
    [1, 3, 5, 2, 4, 6]
    >>> merge_sorted([], [1, 2, 3])
    []
    """
    # Two indices, one for each list
    i, j = 0, 0
    # Result list
    result = []

    # While we have elements in both lists, compare and append the smaller one
    while i < len(a) and j < len(b):
        if a[i] <= b[j]:
            result.append(a[i])
            i += 1
        else:
            result.append(b[j])
            j += 1

    # Append any remaining elements from list `a`
    while i < len(a):
        result.append(a[i])
        i += 1

    # Append any remaining elements from list `b`
    while j < len(b):
        result.append(b[j])
        j += 1

    return result
