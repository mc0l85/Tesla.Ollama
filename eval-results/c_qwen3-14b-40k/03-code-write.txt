def merge_sorted(list1, list2):
    """
    Merges two sorted lists into one sorted list without using the built-in `sorted()` function or `.sort()` method.

    Args:
        list1 (list of int): The first sorted list of integers.
        list2 (list of int): The second sorted list of integers.

    Returns:
        list of int: A new list containing all elements from both input lists, sorted in ascending order.
    """
    merged = []
    i, j = 0, 0

    # Traverse both lists until one is exhausted
    while i < len(list1) and j < len(list2):
        if list1[i] < list2[j]:
            merged.append(list1[i])
            i += 1
        else:
            merged.append(list2[j])
            j += 1

    # Add remaining elements from list1 or list2
    merged.extend(list1[i:])
    merged.extend(list2[j:])

    return merged


# Test cases
assert merge_sorted([1, 3, 5], [2, 4, 6]) == [1, 2, 3, 4, 5, 6]
assert merge_sorted([], [1, 2, 3]) == [1, 2, 3]
